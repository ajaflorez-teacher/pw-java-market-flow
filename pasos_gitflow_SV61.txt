# Clonar el repositorio y inicializar el GitFlow

git clone https://github.com/ajaflorez-teacher/pw-java-market-flow.git
cd pw-java-market-flow
git fetch origin develop
git flow init
git checkout develop

# Antes de crear un feature 
# Antes de subir a develop 
# Antes de hacer un git flow feature finish

git pull origin develop

# Gestionando un feature

git flow feature start HU99_NombreApellido
git branch

# Editar codigo en un IDE

git add .
git commit -m "Added JuanSucursal Class"
git pull origin develop:develop
git flow feature publish HU11_JuanSucarsales

# En el caso de que tenga que realizar una actualizaci√≥n del feature publicado, eliminar el branch remoto
git push -D origin feature/HU17_JuanFlores17
# Modificar el codigo, y luego hace git add, git commit y nuevamente git flow feature publish

# POdemos descargar el cogido para verficarlo

# Si todo esta OK, merge(fusionar) a nuestro codigo develop

# Si sale ERROR, se debe de actualizar el develop
git pull origin develop:develop
git flow feature finish HU11_JuanSucarsales
git push origin develop:develop
